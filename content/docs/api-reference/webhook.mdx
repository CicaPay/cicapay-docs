## Overview

Webhooks let our platform notify your server in real time whenever an event occurs — a payment received, a payout completed, or a transaction confirmed. Instead of polling our API, you register a URL and we `POST` a signed JSON payload to it automatically.

<Cards>
  <Card icon={<Zap />} title="Real-time">
    Events are delivered within seconds of occurrence.
  </Card>
  <Card icon={<Shield />} title="Secure">
    Every payload is HMAC-SHA256 signed. You verify every request.
  </Card>
  <Card icon={<RefreshCw />} title="Reliable">
    Failed deliveries are retried automatically up to 5 times.
  </Card>

## Step 2 — Understand the Request Format

Every webhook request we send is a standard `HTTP POST` with the following headers:

```http
Content-Type: application/json
X-Signature: <hmac-sha256-hex>
X-Event: <event-name>
```

| Header | Description |
|---|---|
| `Content-Type` | Always `application/json`. |
| `X-Signature` | HMAC-SHA256 hex digest of the full JSON body, signed with your secret key. |
| `X-Event` | The name of the event (e.g. `payment.paid`). |

### Payload Structure

Every payload follows this envelope:

```json
{
  "event": "payment.paid",
  "timestamp": 1773529966,
  "data": { ... }
}
```

| Field | Type | Description |
|---|---|---|
| `event` | `string` | The event type. |
| `timestamp` | `number` | Unix timestamp (seconds) of when the event was emitted. |
| `data` | `object` | Event-specific payload. See [Event Payloads](#event-payloads). |

## Step 3 — Verify the Signature

<Callout title="Always Verify" type="warn">
**Never process a webhook without first verifying its signature.** Skipping this step exposes your system to spoofed requests.
</Callout>

### How We Sign

Our server builds the payload object, serialises it to JSON, and signs it with your secret:

```ts
// Server-side (for reference only)
const payload = {
  event,
  timestamp: Math.floor(Date.now() / 1000),
  data,
};

const signature = crypto
  .createHmac("sha256", secret)
  .update(JSON.stringify(payload))
  .digest("hex");
```

The resulting hex string is sent in the `X-Signature` header.

### How You Verify

Your handler must:

1. Read the raw JSON body **before** parsing it.
2. Recompute the HMAC-SHA256 of the body using your secret key.
3. Compare the result to the `X-Signature` header using a constant-time comparison.
4. Reject the request if the signatures do not match.

<Callout type="error" title="Use Constant-Time Comparison">
A regular `===` comparison is vulnerable to timing attacks. Always use your language's built-in constant-time comparison function (e.g. `crypto.timingSafeEqual` in Node.js).
</Callout>

### Signature Verification Examples

<Tabs items={['Node.js', 'Python', 'PHP', 'Go']}>

<Tab>

```ts title="webhook.ts"
import crypto from "crypto";
import { IncomingMessage, ServerResponse } from "http";

function verifySignature(rawBody: string, secret: string, signature: string): boolean {
  const expected = crypto
    .createHmac("sha256", secret)
    .update(rawBody)
    .digest("hex");

  // Constant-time comparison to prevent timing attacks
  return crypto.timingSafeEqual(
    Buffer.from(expected, "hex"),
    Buffer.from(signature, "hex")
  );
}

// Express example
app.post("/webhook", express.raw({ type: "application/json" }), (req, res) => {
  const rawBody = req.body.toString("utf8");
  const signature = req.headers["x-signature"] as string;
  const secret = process.env.WEBHOOK_SECRET!;

  if (!verifySignature(rawBody, secret, signature)) {
    return res.status(401).json({ error: "Invalid signature" });
  }

  const payload = JSON.parse(rawBody);
  console.log("Received event:", payload.event);

  // Handle event...
  res.status(200).json({ received: true });
});
```

```python title="webhook.py"
import hmac
import hashlib
import json
from flask import Flask, request, jsonify
import os

app = Flask(__name__)

def verify_signature(raw_body: str, secret: str, signature: str) -> bool:
    expected = hmac.new(
        secret.encode("utf-8"),
        raw_body.encode("utf-8"),
        hashlib.sha256
    ).hexdigest()

    # Constant-time comparison
    return hmac.compare_digest(expected, signature)

@app.route("/webhook", methods=["POST"])
def webhook():
    raw_body = request.get_data(as_text=True)
    signature = request.headers.get("X-Signature", "")
    secret = os.environ["WEBHOOK_SECRET"]

    if not verify_signature(raw_body, secret, signature):
        return jsonify({"error": "Invalid signature"}), 401

    payload = json.loads(raw_body)
    print(f"Received event: {payload['event']}")

    # Handle event...
    return jsonify({"received": True}), 200
```

```go title="webhook.go"
package main

import (
	"crypto/hmac"
	"crypto/sha256"
	"encoding/hex"
	"encoding/json"
	"io"
	"log"
	"net/http"
	"os"
)

func verifySignature(rawBody []byte, secret, signature string) bool {
	mac := hmac.New(sha256.New, []byte(secret))
	mac.Write(rawBody)
	expected := hex.EncodeToString(mac.Sum(nil))
	// Constant-time comparison
	return hmac.Equal([]byte(expected), []byte(signature))
}

func webhookHandler(w http.ResponseWriter, r *http.Request) {
	rawBody, err := io.ReadAll(r.Body)
	if err != nil {
		http.Error(w, "Cannot read body", http.StatusBadRequest)
		return
	}

	signature := r.Header.Get("X-Signature")
	secret := os.Getenv("WEBHOOK_SECRET")

	if !verifySignature(rawBody, secret, signature) {
		http.Error(w, `{"error":"Invalid signature"}`, http.StatusUnauthorized)
		return
	}

	var payload map[string]interface{}
	json.Unmarshal(rawBody, &payload)
	log.Printf("Received event: %v", payload["event"])

	// Handle event...
	w.Header().Set("Content-Type", "application/json")
	w.Write([]byte(`{"received":true}`))
}

func main() {
	http.HandleFunc("/webhook", webhookHandler)
	http.ListenAndServe(":8080", nil)
}
```

## Step 4 — Respond Correctly

Your endpoint **must** return a `2xx` HTTP status code within **5 seconds** to acknowledge receipt. Our system considers any other response (or a timeout) a delivery failure.

<Callout title="Respond First, Process Later">
Do not perform heavy processing synchronously inside the webhook handler. Acknowledge the request immediately with `200 OK`, then process the event asynchronously (e.g. via a job queue).
</Callout>

```ts
// ✅ Correct pattern
app.post("/webhook", (req, res) => {
  // 1. Verify signature
  // 2. Acknowledge immediately
  res.status(200).json({ received: true });

  // 3. Process asynchronously
  processEvent(req.body).catch(console.error);
});
```

## Retry Policy & Dead Letter Queue

If our server does not receive a `2xx` response (network error, timeout, `4xx`, `5xx`), the delivery is retried automatically.

### Retry Timeline

```
Attempt 1  ──────────────────────────────►  T+0s   (initial delivery)
                  ↓ failure
Attempt 2  ──────────────────────────────►  T+30s (±6s)
                  ↓ failure
Attempt 3  ──────────────────────────────►  T+1min30s (±18s)
                  ↓ failure
Attempt 4  ──────────────────────────────►  T+3min30s (±42s)
                  ↓ failure
Attempt 5  ──────────────────────────────►  T+7min30s (±1min30s)
                  ↓ failure
Dead Letter Queue (DLQ) — webhook DISABLED
```

| Attempt | Delay after previous failure |
|---|---|
| 1 | Immediate |
| 2 | ~30 seconds |
| 3 | ~2 minutes |
| 4 | ~4 minutes |
| 5 | ~8 minutes |
| DLQ | — webhook disabled |

<Callout type="error" title="Dead Letter Queue">
After **5 consecutive failures**, your webhook endpoint is added to the **Dead Letter Queue** and automatically disabled. It will **no longer receive any notifications** until you re-enable it from the Dashboard.

To re-enable: Dashboard → Webhooks → find the disabled endpoint → click **Re-enable**.
</Callout>

## Security Best Practices

<Callout type="idea" title="Security Checklist">
Run through this checklist when deploying your webhook handler.
</Callout>

### ✅ Always Verify the Signature

Never skip signature verification, even in development. A missing or invalid `X-Signature` must result in a `401 Unauthorized` response — not silent acceptance.

### ✅ Use HTTPS Only

Your listener URL must be a valid `https://` endpoint. Plaintext `http://` URLs are rejected. HTTPS ensures the payload cannot be read or tampered with in transit.

### ✅ Store Your Secret in Environment Variables

```bash
# .env
WEBHOOK_SECRET=your_strong_random_secret

# Never do this ❌
const secret = "hardcoded_secret_in_code";
```

### ✅ Protect Against Replay Attacks

The `timestamp` field in every payload is a Unix timestamp. Reject requests where the timestamp is more than 5 minutes old:

```ts
const TOLERANCE_SECONDS = 5 * 60; // 5 minutes

const payload = JSON.parse(rawBody);
const age = Math.floor(Date.now() / 1000) - payload.timestamp;

if (age > TOLERANCE_SECONDS) {
  return res.status(400).json({ error: "Stale request" });
}
```

### ✅ Return 200 Quickly

Respond within 5 seconds. Offload all processing to a background queue to avoid timeouts that trigger unnecessary retries.

### ✅ Make Your Handler Idempotent

Retries mean you may receive the same event more than once. Use the combination of `event` + a unique ID from `data` (e.g. `payout_id`, `payment_id`) as a deduplication key:

```ts
const dedupKey = `${payload.event}:${payload.data.payout_id}`;
if (await redis.exists(dedupKey)) {
  return res.status(200).json({ received: true }); // already processed
}
await redis.set(dedupKey, 1, "EX", 86400);
// ... process event
```

### ✅ Monitor Your Endpoint

Watch your server logs and set up alerts for `401` and `5xx` responses on your webhook route. A disabled webhook (DLQ) means missed events.

## Troubleshooting

| Symptom | Likely Cause | Solution |
|---|---|---|
| `401 Unauthorized` from your server | Signature mismatch | Check that you're using the raw body string, not a re-serialised object. |
| Webhook appears disabled | 5 consecutive failures → DLQ | Fix your endpoint, then re-enable from Dashboard. |
| Test succeeds but live events don't arrive | Event not subscribed | Check your subscribed events list in Dashboard. |
| Duplicate events received | Normal retry behaviour | Implement idempotency using event + resource ID. |
| Timeout on your side | Slow synchronous processing | Acknowledge with `200` immediately, process async. |
